1//! Byte-class authority: bounded `u8` state alphabets and the admission status vocabulary.
2//!
3//! The whole crate operates on small, bounded `u8` domains so that kernels remain
4//! branchless and SIMD-friendly. Joining and clamping lower onto [`bcinr_logic`]
5//! primitives ([`bcinr_logic::mask::max_u32`], [`bcinr_logic::fix::clamp_u32`]) rather
6//! than `if`/`match`.
7//!
8//! # Status lattice
9//!
10//! [`status`] constants form a total order whose numeric value is the "severity" of the
11//! observation. This lets a branchless `max` compute the worst-of join across an event
12//! window. Terminal/abnormal codes (`REFUSED`, `RESIDUAL`) sort above the normal lifecycle
13//! so they dominate a join.
14//!
15//! ```text
16//! UNKNOWN(0) < BLOCKED(1) < PARTIAL(2) < PENDING(3) < ADMITTED(4)
17//!           < PROJECTED(5) < RECEIPTED(6) < REFUSED(7) < RESIDUAL(8)
18//! ```
19

21use bcinr_logic::mask;

23/// The admission status vocabulary, as ordered `u8` codes.
24///
25/// Ordering is meaningful: it forms a coarse "admission lattice" so a worst-of join can
26/// be computed with a branchless `max`. Terminal/abnormal codes (`REFUSED`, `RESIDUAL`)
27/// sort above the normal lifecycle so they dominate a join.
28///
29/// All constants are in `[0, COUNT)`. `COUNT` itself is not a valid status code; it is
30/// the cardinality of the vocabulary for bounds-checking.
31pub mod status {
32    /// Nothing is known yet. The zero/default state.
33    pub const UNKNOWN: u8 = 0;
34    /// A precondition prevented progress.
35    pub const BLOCKED: u8 = 1;
36    /// Partially observed; not yet admissible.
37    pub const PARTIAL: u8 = 2;
38    /// Awaiting a downstream decision.
39    pub const PENDING: u8 = 3;
40    /// Admitted into bounded state.
41    pub const ADMITTED: u8 = 4;
42    /// Projected to an engine/host surface.
43    pub const PROJECTED: u8 = 5;
44    /// Sealed into a receipt chain.
45    pub const RECEIPTED: u8 = 6;
46    /// Refused by an admission rule.
47    pub const REFUSED: u8 = 7;
48    /// Residual diagnostic preserved after repair.
49    pub const RESIDUAL: u8 = 8;
50    /// Cardinality of the vocabulary (number of distinct codes).
51    ///
52    /// All valid status codes satisfy `code < COUNT`.
53    pub const COUNT: u8 = 9;
54
55    /// Return the human-readable name of a status code, or `"?"` for out-of-range values.
56    ///
57    /// # Examples
58    ///
59    /// ```
60    /// use wasm4games::class::status;
61    /// assert_eq!(status::name(status::ADMITTED), "ADMITTED");
62    /// assert_eq!(status::name(99), "?");
63    /// ```
64    #[must_use]
65    pub const fn name(code: u8) -> &'static str {
66        match code {
67            UNKNOWN => "UNKNOWN",
68            BLOCKED => "BLOCKED",
69            PARTIAL => "PARTIAL",
70            PENDING => "PENDING",
71            ADMITTED => "ADMITTED",
72            PROJECTED => "PROJECTED",
73            RECEIPTED => "RECEIPTED",
74            REFUSED => "REFUSED",
75            RESIDUAL => "RESIDUAL",
76            _ => "?",
77        }
78    }
79
80    /// Return `true` if `code` is a member of the vocabulary (`code < COUNT`).
81    ///
82    /// # Examples
83    ///
84    /// ```
85    /// use wasm4games::class::status;
86    /// assert!(status::is_valid(status::REFUSED));
87    /// assert!(!status::is_valid(status::COUNT));
88    /// assert!(!status::is_valid(99));
89    /// ```
90    #[must_use]
91    pub const fn is_valid(code: u8) -> bool {
92        code < COUNT
93    }
94}

96/// A bounded byte class: a `u8` drawn from a domain of known cardinality.
97///
98/// `ByteClass` is a newtype over `u8` that makes the bounded domain explicit.
99/// The `Default` implementation returns `ByteClass(0)`, which is the first
100/// (lowest) class in any alphabet.
101///
102/// # Examples
103///
104/// ```
105/// use wasm4games::class::{ByteClass, status};
106///
107/// // Clamp an out-of-range value into the status vocabulary
108/// let raw = ByteClass(200);
109/// let clamped = raw.clamp(status::COUNT);
110/// assert_eq!(clamped.raw(), status::COUNT - 1);
111///
112/// // Within-range values pass through unchanged
113/// assert_eq!(ByteClass(3).clamp(status::COUNT).raw(), 3);
114/// ```
115#[derive(Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Debug, Default)]
116pub struct ByteClass(pub u8);
117
118impl ByteClass {
119    /// Construct a [`ByteClass`] from a raw `u8` code.
120    ///
121    /// # Examples
122    ///
123    /// ```
124    /// use wasm4games::class::ByteClass;
125    /// assert_eq!(ByteClass::new(5).raw(), 5);
126    /// ```
127    #[inline]
128    #[must_use]
129    pub const fn new(code: u8) -> Self {
130        Self(code)
131    }
132
133    /// The raw `u8` code.
134    ///
135    /// # Examples
136    ///
137    /// ```
138    /// use wasm4games::class::ByteClass;
139    /// assert_eq!(ByteClass(7).raw(), 7);
140    /// ```
141    #[inline]
142    #[must_use]
143    pub const fn raw(self) -> u8 {
144        self.0
145    }
146
147    /// Return `true` if this class is within the domain `[0, card)`.
148    ///
149    /// # Examples
150    ///
151    /// ```
152    /// use wasm4games::class::{ByteClass, status};
153    /// assert!(ByteClass(3).is_in_domain(status::COUNT));
154    /// assert!(!ByteClass(200).is_in_domain(status::COUNT));
155    /// ```
156    #[inline]
157    #[must_use]
158    pub const fn is_in_domain(self, card: u8) -> bool {
159        self.0 < card
160    }
161
162    /// Branchlessly clamp this class into the domain `[0, card)`.
163    ///
164    /// A `card` of `0` is treated as `1` (the domain `{0}`), so the result is always
165    /// `0` in that degenerate case. This matches the semantics of
166    /// [`bcinr_logic::fix::clamp_u32`] which is used for the implementation.
167    ///
168    /// # Examples
169    ///
170    /// ```
171    /// use wasm4games::class::{ByteClass, status};
172    ///
173    /// assert_eq!(ByteClass(200).clamp(status::COUNT).raw(), status::COUNT - 1);
174    /// assert_eq!(ByteClass(3).clamp(status::COUNT).raw(), 3);
175    /// assert_eq!(ByteClass(5).clamp(0).raw(), 0);
176    /// ```
177    #[inline]
178    #[must_use]
179    pub fn clamp(self, card: u8) -> ByteClass {
180        let hi = card.saturating_sub(1) as u32;
181        ByteClass(fix::clamp_u32(self.0 as u32, 0, hi) as u8)
182    }
183}
184
185impl core::fmt::Display for ByteClass {
186    #[inline]
187    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
188        write!(f, "ByteClass({})", self.0)
189    }
190}

192/// An admission [`status`] code as a typed value.
193///
194/// `Status` is a newtype over `u8` that wraps a code from the [`status`] vocabulary.
195/// The `Default` implementation returns `Status(0)` which corresponds to
196/// [`status::UNKNOWN`] — the "nothing known yet" state.
197///
198/// # Lattice semantics
199///
200/// Status codes form a total order. The [`join`](Self::join) method computes the
201/// worst-of (max-code) combination of two statuses branchlessly using
202/// [`bcinr_logic::mask::max_u32`], so an `ADMITTED` joined with `REFUSED` yields
203/// `REFUSED` without any branch.
204///
205/// # Examples
206///
207/// ```
208/// use wasm4games::class::{Status, status};
209///
210/// let a = Status::new(status::ADMITTED);
211/// let b = Status::new(status::REFUSED);
212/// assert_eq!(a.join(b).raw(), status::REFUSED);
213/// assert_eq!(a.to_string(), "Status(ADMITTED)");
214/// ```
215#[derive(Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Debug, Default)]
216pub struct Status(pub u8);
217
218impl Status {
219    /// The `UNKNOWN` status: nothing is known yet. This is also the [`Default`] value.
220    pub const UNKNOWN: Self = Self(status::UNKNOWN);
221    /// The `BLOCKED` status: a precondition prevented progress.
222    pub const BLOCKED: Self = Self(status::BLOCKED);
223    /// The `PARTIAL` status: partially observed; not yet admissible.
224    pub const PARTIAL: Self = Self(status::PARTIAL);
225    /// The `PENDING` status: awaiting a downstream decision.
226    pub const PENDING: Self = Self(status::PENDING);
227    /// The `ADMITTED` status: admitted into bounded state.
228    pub const ADMITTED: Self = Self(status::ADMITTED);
229    /// The `PROJECTED` status: projected to an engine/host surface.
230    pub const PROJECTED: Self = Self(status::PROJECTED);
231    /// The `RECEIPTED` status: sealed into a receipt chain.
232    pub const RECEIPTED: Self = Self(status::RECEIPTED);
233    /// The `REFUSED` status: refused by an admission rule.
234    pub const REFUSED: Self = Self(status::REFUSED);
235    /// The `RESIDUAL` status: residual diagnostic preserved after repair.
236    pub const RESIDUAL: Self = Self(status::RESIDUAL);
237
238    /// Construct a [`Status`] from a raw `u8` code.
239    ///
240    /// The value is not validated against [`status::COUNT`]; use
241    /// [`is_valid`](Self::is_valid) if you need a bounds check.
242    ///
243    /// # Examples
244    ///
245    /// ```
246    /// use wasm4games::class::{Status, status};
247    /// assert_eq!(Status::new(status::ADMITTED).raw(), status::ADMITTED);
248    /// ```
249    #[inline]
250    #[must_use]
251    pub const fn new(code: u8) -> Self {
252        Self(code)
253    }
254
255    /// Return the raw `u8` code.
256    ///
257    /// # Examples
258    ///
259    /// ```
260    /// use wasm4games::class::{Status, status};
261    /// assert_eq!(Status::new(status::REFUSED).raw(), 7);
262    /// ```
263    #[inline]
264    #[must_use]
265    pub const fn raw(self) -> u8 {
266        self.0
267    }
268
269    /// Return `true` if this status code is in the vocabulary (`self.0 < status::COUNT`).
270    ///
271    /// # Examples
272    ///
273    /// ```
274    /// use wasm4games::class::{Status, status};
275    /// assert!(Status::new(status::RESIDUAL).is_valid());
276    /// assert!(!Status::new(status::COUNT).is_valid());
277    /// ```
278    #[inline]
279    #[must_use]
280    pub const fn is_valid(self) -> bool {
281        status::is_valid(self.0)
282    }
283
284    /// Return the human-readable name of this status code.
285    ///
286    /// Delegates to [`status::name`]. Returns `"?"` for out-of-vocabulary codes.
287    ///
288    /// # Examples
289    ///
290    /// ```
291    /// use wasm4games::class::{Status, status};
292    /// assert_eq!(Status::ADMITTED.status_name(), "ADMITTED");
293    /// assert_eq!(Status::new(99).status_name(), "?");
294    /// ```
295    #[inline]
296    #[must_use]
297    pub const fn status_name(self) -> &'static str {
298        status::name(self.0)
299    }
300
301    /// Worst-of (max-code) lattice join, computed branchlessly.
302    ///
303    /// Uses [`bcinr_logic::mask::max_u32`] so there is no branch in the machine code.
304    /// Because status codes are ordered by severity, this always returns the "worse"
305    /// of the two statuses.
306    ///
307    /// # Examples
308    ///
309    /// ```
310    /// use wasm4games::class::{Status, status};
311    ///
312    /// assert_eq!(
313    ///     Status::new(status::ADMITTED).join(Status::new(status::REFUSED)).raw(),
314    ///     status::REFUSED,
315    /// );
316    /// assert_eq!(
317    ///     Status::new(status::UNKNOWN).join(Status::new(status::ADMITTED)).raw(),
318    ///     status::ADMITTED,
319    /// );
320    /// // Join is commutative
321    /// assert_eq!(
322    ///     Status::PARTIAL.join(Status::REFUSED),
323    ///     Status::REFUSED.join(Status::PARTIAL),
324    /// );
325    /// ```
326    #[inline]
327    #[must_use]
328    pub fn join(self, other: Status) -> Status {
329        Status(mask::max_u32(self.0 as u32, other.0 as u32) as u8)
330    }
331}
332
333impl core::fmt::Display for Status {
334    #[inline]
335    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
336        write!(f, "Status({})", status::name(self.0))
337    }
338}

347    #[test]
348    fn status_vocabulary_and_byte_class() {
349        // Lattice order
350        assert!(status::UNKNOWN < status::BLOCKED);
351        assert!(status::BLOCKED < status::PARTIAL);
352        assert!(status::PARTIAL < status::PENDING);
353        assert!(status::PENDING < status::ADMITTED);
354        assert!(status::ADMITTED < status::PROJECTED);
355        assert!(status::PROJECTED < status::RECEIPTED);
356        assert!(status::RECEIPTED < status::REFUSED);
357        assert!(status::REFUSED < status::RESIDUAL);
358        assert!(status::RESIDUAL < status::COUNT);
359
360        // is_valid
361        for code in 0..status::COUNT {
362            assert!(status::is_valid(code), "code {code} should be valid");
363        }
364        assert!(!status::is_valid(status::COUNT));
365        assert!(!status::is_valid(99));
366
367        // name table
368        let expected: &[(u8, &str)] = &[
369            (status::UNKNOWN, "UNKNOWN"),
370            (status::BLOCKED, "BLOCKED"),
371            (status::PARTIAL, "PARTIAL"),
372            (status::PENDING, "PENDING"),
373            (status::ADMITTED, "ADMITTED"),
374            (status::PROJECTED, "PROJECTED"),
375            (status::RECEIPTED, "RECEIPTED"),
376            (status::REFUSED, "REFUSED"),
377            (status::RESIDUAL, "RESIDUAL"),
378        ];
379        for &(code, name) in expected {
380            assert_eq!(status::name(code), name, "wrong name for code {code}");
381        }
382        assert_eq!(status::name(99), "?");
383
384        // ByteClass: clamp, is_in_domain, Display, default, Ord, raw
385        assert_eq!(ByteClass(200).clamp(status::COUNT).raw(), status::COUNT - 1);
386        assert_eq!(ByteClass(3).clamp(status::COUNT).raw(), 3);
387        assert_eq!(ByteClass(5).clamp(0).raw(), 0);
388        assert_eq!(ByteClass(8).clamp(9).raw(), 8);
389        assert_eq!(ByteClass(9).clamp(9).raw(), 8);
390        assert!(ByteClass(0).is_in_domain(status::COUNT));
391        assert!(ByteClass(8).is_in_domain(status::COUNT));
392        assert!(!ByteClass(9).is_in_domain(status::COUNT));
393        assert!(!ByteClass(200).is_in_domain(status::COUNT));
394        assert_eq!(ByteClass(3).to_string(), "ByteClass(3)");
395        assert_eq!(ByteClass::default().raw(), 0);
396        assert_eq!(ByteClass::default(), ByteClass::new(0));
397        assert!(ByteClass(0) < ByteClass(1));
398        let mut classes = [ByteClass(5), ByteClass(1), ByteClass(3)];
399        classes.sort();
400        assert_eq!(classes, [ByteClass(1), ByteClass(3), ByteClass(5)]);
401        for v in [0u8, 1, 8, 127, 255] {
402            assert_eq!(ByteClass::new(v).raw(), v);
403        }
404    }

406    #[test]
407    fn status_typed_behavior() {
408        // join: worst-of, commutative, idempotent
409        assert_eq!(
410            Status(status::ADMITTED).join(Status(status::REFUSED)).raw(),
411            status::REFUSED
412        );
413        assert_eq!(
414            Status(status::UNKNOWN).join(Status(status::ADMITTED)).raw(),
415            status::ADMITTED
416        );
417        for a in 0..status::COUNT {
418            for b in 0..status::COUNT {
419                assert_eq!(
420                    Status(a).join(Status(b)),
421                    Status(b).join(Status(a)),
422                    "join not commutative for ({a},{b})"
423                );
424            }
425        }
426        for code in 0..status::COUNT {
427            let s = Status(code);
428            assert_eq!(s.join(s), s, "join not idempotent for {code}");
429        }
430
431        // Display
432        assert_eq!(
433            Status::new(status::ADMITTED).to_string(),
434            "Status(ADMITTED)"
435        );
436        assert_eq!(Status::new(status::UNKNOWN).to_string(), "Status(UNKNOWN)");
437        assert_eq!(Status::new(status::REFUSED).to_string(), "Status(REFUSED)");
438        assert_eq!(
439            Status::new(status::RESIDUAL).to_string(),
440            "Status(RESIDUAL)"
441        );
442        assert_eq!(Status::new(99).to_string(), "Status(?)");
443
444        // default, const associated values, is_valid, status_name
445        assert_eq!(Status::default().raw(), status::UNKNOWN);
446        assert_eq!(Status::default(), Status::UNKNOWN);
447        assert_eq!(Status::UNKNOWN.raw(), status::UNKNOWN);
448        assert_eq!(Status::REFUSED.raw(), status::REFUSED);
449        assert_eq!(Status::RESIDUAL.raw(), status::RESIDUAL);
450        for code in 0..status::COUNT {
451            assert!(Status::new(code).is_valid(), "code {code} should be valid");
452        }
453        assert!(!Status::new(status::COUNT).is_valid());
454        assert_eq!(Status::ADMITTED.status_name(), "ADMITTED");
455        assert_eq!(Status::new(99).status_name(), "?");
456
457        // Ord + BTreeMap key
458        assert!(Status::UNKNOWN < Status::BLOCKED);
459        assert!(Status::REFUSED < Status::RESIDUAL);
460        let mut statuses = [Status::REFUSED, Status::UNKNOWN, Status::ADMITTED];
461        statuses.sort();
462        assert_eq!(
463            statuses,
464            [Status::UNKNOWN, Status::ADMITTED, Status::REFUSED]
465        );
466        let mut map = BTreeMap::new();
467        map.insert(Status::ADMITTED, "ok");
468        map.insert(Status::REFUSED, "no");
469        assert_eq!(map[&Status::ADMITTED], "ok");
470        assert_eq!(map[&Status::REFUSED], "no");
471    }
