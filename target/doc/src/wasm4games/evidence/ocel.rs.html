1//! Object-centric event log (OCEL-style) types.
2//!
3//! Default build is allocation-free: an [`OcelEvent`] is `Copy` and links to a bounded set
4//! of objects via [`ObjectRefs`]. An `alloc`-gated [`OcelLog`] (a growable event stream) and
5//! JSON export are added on top of these core types.
6//!
7//! # OCEL model
8//!
9//! In OCEL-2.0, an event is an activity execution at a point in time that is associated with
10//! one or more typed objects. [`OcelEvent`] encodes exactly that: an `event_code` (activity
11//! type), an `activity` (pattern id), a `timestamp` (logical tick), a bounded set of
12//! `(object_type_code, object_id)` pairs ([`ObjectRefs`]), and the admission `status` code.
13//!
14//! # Invariants
15//!
16//! * `ObjectRefs::len() <= ObjectRefs::CAP` at all times.
17//! * `write_to` is deterministic and architecture-independent (big-endian encoding).
18//! * An [`OcelEvent`] with no object links is valid but may indicate a laundering pattern —
19//!   see [`ObjectRefs::is_empty`].
20

23/// A bounded set of `(object_type_code, object_id)` references for one event.
24///
25/// Fixed capacity keeps [`OcelEvent`] `Copy` and allocation-free in the default `no_std`
26/// build.  Pushing beyond [`Self::CAP`] silently saturates; the excess links are dropped.
27///
28/// # Invariants
29///
30/// `self.len() <= Self::CAP` at all times. There is no way to violate this invariant through
31/// the public API.
32///
33/// # Examples
34///
35/// ```
36/// use wasm4games::evidence::ocel::ObjectRefs;
37/// let mut refs = ObjectRefs::new();
38/// refs.push(1, 42);
39/// refs.push(2, 99);
40/// assert_eq!(refs.len(), 2);
41/// assert_eq!(refs.as_slice(), &[(1, 42), (2, 99)]);
42/// ```
43#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug)]
44pub struct ObjectRefs {
45    items: [(u16, u64); Self::CAP],
46    len: u8,
47}
48
49impl ObjectRefs {
50    /// Maximum number of object references a single event may carry.
51    pub const CAP: usize = 4;
52
53    /// An empty set of object references.
54    ///
55    /// Identical to [`Default::default`]; provided as `const fn` for use in `const` contexts.
56    #[inline]
57    #[must_use = "returns an empty ObjectRefs; discard only if you immediately overwrite it"]
58    pub const fn new() -> Self {
59        Self {
60            items: [(0, 0); Self::CAP],
61            len: 0,
62        }
63    }
64
65    /// Push a `(type_code, id)` reference.
66    ///
67    /// Silently saturates at [`Self::CAP`]; excess pushes are dropped without error.
68    #[inline]
69    pub fn push(&mut self, type_code: u16, id: u64) {
70        let idx = self.len as usize;
71        if idx < Self::CAP {
72            self.items[idx] = (type_code, id);
73            self.len += 1;
74        }
75    }
76
77    /// The live `(type_code, object_id)` references as a slice.
78    #[inline]
79    #[must_use = "returns a slice of the live object references; ignoring it is likely a bug"]
80    pub fn as_slice(&self) -> &[(u16, u64)] {
81        &self.items[..self.len as usize]
82    }
83
84    /// Number of live references.
85    #[inline]
86    #[must_use = "returns the count of live object references"]
87    pub fn len(&self) -> usize {
88        self.len as usize
89    }
90
91    /// Whether there are no references.
92    ///
93    /// An event with no object links is valid but may indicate an OCEL-laundering pattern
94    /// (an event that cannot be traced back to any real object).
95    #[inline]
96    #[must_use = "returns true if there are no object references; check this before trusting the event"]
97    pub fn is_empty(&self) -> bool {
98        self.len == 0
99    }
100}
101
102impl Default for ObjectRefs {
103    #[inline]
104    fn default() -> Self {
105        Self::new()
106    }
107}
108
109impl fmt::Display for ObjectRefs {
110    /// Formats the object references as a bracketed list: `[(type_code, id), ...]`.
111    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
112        f.write_str("[")?;
113        for (i, &(tc, id)) in self.as_slice().iter().enumerate() {
114            if i > 0 {
115                f.write_str(", ")?;
116            }
117            write!(f, "({tc}, {id})")?;
118        }
119        f.write_str("]")
120    }
121}

123/// An object-centric event: an activity at a logical time, linked to real objects.
124///
125/// # Fields
126///
127/// * `event_code` — Event-type code, matches [`crate::ir::EventKind::code`].
128/// * `activity` — Pattern id that produced the event.
129/// * `timestamp` — Logical tick at which the event occurred.  Use [`From<u64>`] to build a
130///   tick wrapper; here the raw `u64` is stored directly.
131/// * `objects` — The [`ObjectRefs`] set linking this event to typed objects.
132/// * `status` — Admission [`crate::class::status`] code at emission time.
133///
134/// # Invariants
135///
136/// The event is `Copy`; all state is inline with no heap pointers.
137///
138/// # Examples
139///
140/// ```
141/// use wasm4games::evidence::ocel::OcelEvent;
142/// let mut ev = OcelEvent::new(0x01, 0x02, 100, 4);
143/// ev.objects.push(0xAA, 9999);
144/// assert_eq!(ev.objects.len(), 1);
145/// assert_eq!(ev.timestamp, 100);
146/// ```
147#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug)]
148pub struct OcelEvent {
149    /// Event-type code (matches [`crate::ir::EventKind::code`]).
150    pub event_code: u16,
151    /// The pattern id that produced the event.
152    pub activity: u16,
153    /// Logical tick at which the event occurred.
154    pub timestamp: u64,
155    /// The objects the event links to.
156    pub objects: ObjectRefs,
157    /// Admission [`crate::class::status`] code at emission time.
158    pub status: u8,
159}
160
161impl OcelEvent {
162    /// Construct an event with no object links yet.
163    ///
164    /// Object references can be pushed after construction via [`ObjectRefs::push`].
165    ///
166    /// # Examples
167    ///
168    /// ```
169    /// use wasm4games::evidence::ocel::OcelEvent;
170    /// let ev = OcelEvent::new(1, 2, 3, 4);
171    /// assert!(ev.objects.is_empty());
172    /// assert_eq!(ev.event_code, 1);
173    /// assert_eq!(ev.activity, 2);
174    /// assert_eq!(ev.timestamp, 3);
175    /// assert_eq!(ev.status, 4);
176    /// ```
177    #[inline]
178    #[must_use = "returns the new OcelEvent; bind it to a variable"]
179    pub const fn new(event_code: u16, activity: u16, timestamp: u64, status: u8) -> Self {
180        Self {
181            event_code,
182            activity,
183            timestamp,
184            objects: ObjectRefs::new(),
185            status,
186        }
187    }
188
189    /// Fixed-size header of the [`Self::write_to`] encoding, in bytes.
190    ///
191    /// Layout: `event_code` (2) + `activity` (2) + `timestamp` (8) + `status` (1) +
192    /// object count (1) = 14 bytes total.
193    pub const HEADER_BYTES: usize = 14;
194
195    /// Bytes used to encode a single object reference: `type_code` (2) + `id` (8) = 10 bytes.
196    pub const OBJECT_BYTES: usize = 10;
197
198    /// Exact number of bytes [`Self::write_to`] will write for this event.
199    ///
200    /// Equal to [`Self::HEADER_BYTES`] + `self.objects.len()` × [`Self::OBJECT_BYTES`].
201    #[inline]
202    #[must_use = "returns the exact byte count needed for write_to; use it to pre-size your buffer"]
203    pub fn encoded_len(&self) -> usize {
204        Self::HEADER_BYTES + self.objects.len() * Self::OBJECT_BYTES
205    }
206
207    /// Serialize this event into `buf` using a compact, deterministic big-endian encoding,
208    /// returning the number of bytes written.
209    ///
210    /// The encoding is allocation-free and self-describing enough to round-trip the scalar
211    /// fields and every object reference. All multi-byte integers are big-endian so the byte
212    /// stream is identical across architectures (a prerequisite for cross-host receipts).
213    ///
214    /// Returns [`EvidenceError::BufferTooSmall`] without writing if `buf` cannot hold the
215    /// full record (see [`Self::encoded_len`]).
216    ///
217    /// # Examples
218    ///
219    /// ```
220    /// use wasm4games::evidence::ocel::OcelEvent;
221    /// let ev = OcelEvent::new(0x0102, 0x0304, 0x05, 4);
222    /// let mut buf = [0u8; 64];
223    /// let n = ev.write_to(&mut buf).unwrap();
224    /// assert_eq!(n, ev.encoded_len());
225    /// ```
226    #[must_use = "returns the number of bytes written; check for EvidenceError::BufferTooSmall"]
227    pub fn write_to(&self, buf: &mut [u8]) -> Result<usize, EvidenceError> {
228        let need = self.encoded_len();
229        if buf.len() < need {
230            return Err(EvidenceError::BufferTooSmall);
231        }
232        let mut at = 0usize;
233        // Local helper keeps the cursor logic in one place and avoids intermediate allocs.
234        let put_u16 = |buf: &mut [u8], at: &mut usize, v: u16| {
235            buf[*at..*at + 2].copy_from_slice(&v.to_be_bytes());
236            *at += 2;
237        };
238        put_u16(buf, &mut at, self.event_code);
239        put_u16(buf, &mut at, self.activity);
240        buf[at..at + 8].copy_from_slice(&self.timestamp.to_be_bytes());
241        at += 8;
242        buf[at] = self.status;
243        at += 1;
244        let refs = self.objects.as_slice();
245        buf[at] = refs.len() as u8;
246        at += 1;
247        for &(type_code, id) in refs {
248            buf[at..at + 2].copy_from_slice(&type_code.to_be_bytes());
249            at += 2;
250            buf[at..at + 8].copy_from_slice(&id.to_be_bytes());
251            at += 8;
252        }
253        Ok(at)
254    }
255}

268/// An error produced while emitting or serializing evidence.
269///
270/// # Variants
271///
272/// * [`EvidenceError::BufferTooSmall`] — The caller-supplied buffer was shorter than the
273///   encoded record. Pre-size the buffer using [`OcelEvent::encoded_len`].
274#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug)]
275pub enum EvidenceError {
276    /// The destination buffer was too small to hold the encoded record.
277    BufferTooSmall,
278}
279
280impl fmt::Display for EvidenceError {
281    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
282        match self {
283            EvidenceError::BufferTooSmall => {
284                f.write_str("evidence error: buffer too small for encoded record")
285            }
286        }
287    }
288}

290/// Convert a raw `u16` event code into an [`OcelEvent`] with zeroed activity, timestamp, and
291/// status.
292///
293/// Useful as a quick constructor when only the event-type code matters.
294///
295/// # Examples
296///
297/// ```
298/// use wasm4games::evidence::ocel::OcelEvent;
299/// let ev = OcelEvent::from(0x0042u16);
300/// assert_eq!(ev.event_code, 0x0042);
301/// assert_eq!(ev.activity, 0);
302/// assert_eq!(ev.timestamp, 0);
303/// assert_eq!(ev.status, 0);
304/// ```
305impl From<u16> for OcelEvent {
306    #[inline]
307    fn from(event_code: u16) -> Self {
308        Self::new(event_code, 0, 0, 0)
309    }

312/// A growable, object-centric event stream.
313///
314/// Available only with the `alloc` feature: it owns a heap-backed [`alloc::vec::Vec`] of
315/// [`OcelEvent`]s. The default `no_std` build stays allocation-free and uses the bounded
316/// [`OcelEvent`] / [`ObjectRefs`] types directly.
317///
318/// # Examples
319///
320/// ```
321/// # #[cfg(feature = "alloc")] {
322/// use wasm4games::evidence::ocel::{OcelEvent, OcelLog};
323/// let mut log = OcelLog::new();
324/// log.push(OcelEvent::new(1, 2, 3, 4));
325/// assert_eq!(log.len(), 1);
326/// # }
327/// ```
328#[cfg(feature = "alloc")]
329#[derive(Clone, Debug, Default)]
330pub struct OcelLog {
331    events: alloc::vec::Vec<OcelEvent>,
332}
333
334#[cfg(feature = "alloc")]
335impl OcelLog {
336    /// An empty log.
337    ///
338    /// Identical to [`Default::default`]; provided for discoverability.
339    #[inline]
340    #[must_use = "returns an empty OcelLog; bind it to a variable"]
341    pub fn new() -> Self {
342        Self {
343            events: alloc::vec::Vec::new(),
344        }
345    }
346
347    /// Append an event to the stream.
348    #[inline]
349    pub fn push(&mut self, event: OcelEvent) {
350        self.events.push(event);
351    }
352
353    /// Number of events in the log.
354    #[inline]
355    #[must_use = "returns the event count; ignoring it is likely a bug"]
356    pub fn len(&self) -> usize {
357        self.events.len()
358    }
359
360    /// Whether the log holds no events.
361    #[inline]
362    #[must_use = "returns true if the log is empty; check before iterating"]
363    pub fn is_empty(&self) -> bool {
364        self.events.is_empty()
365    }
366
367    /// The events as a slice, in emission order.
368    #[inline]
369    #[must_use = "returns a slice of events; ignoring it discards the reference"]
370    pub fn as_slice(&self) -> &[OcelEvent] {
371        &self.events
372    }
373
374    /// Iterate over the events in emission order.
375    #[inline]
376    pub fn iter(&self) -> core::slice::Iter<'_, OcelEvent> {
377        self.events.iter()
378    }
379
380    /// Render the log as an OCEL-flavored JSON array.
381    ///
382    /// Object-centric: every event carries its own `objects` array of
383    /// `{ "type": <code>, "id": <id> }` links, so the relation between events and objects is
384    /// explicit rather than reconstructed. The JSON is hand-rolled (no `serde`) to keep the
385    /// crate dependency-free; all values are integers, so no string escaping is required.
386    ///
387    /// # Examples
388    ///
389    /// ```
390    /// # #[cfg(feature = "alloc")] {
391    /// use wasm4games::evidence::ocel::{OcelEvent, OcelLog};
392    /// let mut log = OcelLog::new();
393    /// log.push(OcelEvent::new(1, 2, 3, 4));
394    /// let json = log.to_json();
395    /// assert!(json.starts_with('['));
396    /// # }
397    /// ```
398    #[must_use = "returns an owned JSON string; bind or use it"]
399    pub fn to_json(&self) -> alloc::string::String {
400        use core::fmt::Write as _;
401        let mut s = alloc::string::String::new();
402        s.push('[');
403        for (i, ev) in self.events.iter().enumerate() {
404            if i > 0 {
405                s.push(',');
406            }
407            // Scalar fields. `write!` to a String is infallible, hence the `let _ =`.
408            let _ = write!(
409                s,
410                "{{\"event_code\":{},\"activity\":{},\"timestamp\":{},\"status\":{},\"objects\":[",
411                ev.event_code, ev.activity, ev.timestamp, ev.status
412            );
413            for (j, &(type_code, id)) in ev.objects.as_slice().iter().enumerate() {
414                if j > 0 {
415                    s.push(',');
416                }
417                let _ = write!(s, "{{\"type\":{},\"id\":{}}}", type_code, id);
418            }
419            s.push_str("]}");
420        }
421        s.push(']');
422        s
423    }
424}

441    #[test]
442    fn object_refs_behavior() {
443        // empty / default
444        let r = ObjectRefs::new();
445        assert!(r.is_empty());
446        assert_eq!(r.len(), 0);
447        assert_eq!(r.as_slice(), &[]);
448        assert_eq!(ObjectRefs::default(), ObjectRefs::new());
449
450        // push and retrieve
451        let mut r = ObjectRefs::new();
452        r.push(10, 100);
453        r.push(20, 200);
454        assert_eq!(r.len(), 2);
455        assert!(!r.is_empty());
456        assert_eq!(r.as_slice(), &[(10, 100), (20, 200)]);
457
458        // saturates at CAP
459        let mut r = ObjectRefs::new();
460        for i in 0..ObjectRefs::CAP + 5 {
461            r.push(i as u16, i as u64);
462        }
463        assert_eq!(r.len(), ObjectRefs::CAP);
464
465        // Hash is consistent
466        use core::hash::{Hash, Hasher};
467        struct SimpleHasher(u64);
468        impl Hasher for SimpleHasher {
469            fn finish(&self) -> u64 {
470                self.0
471            }
472            fn write(&mut self, bytes: &[u8]) {
473                for &b in bytes {
474                    self.0 = self.0.wrapping_mul(31).wrapping_add(b as u64);
475                }
476            }
477        }
478        let mut r = ObjectRefs::new();
479        r.push(7, 42);
480        let mut h1 = SimpleHasher(0);
481        let mut h2 = SimpleHasher(0);
482        r.hash(&mut h1);
483        r.hash(&mut h2);
484        assert_eq!(h1.finish(), h2.finish());
485    }

487    #[test]
488    fn ocel_event_construction_and_encoding() {
489        // new / From<u16> / multi-object
490        let ev = OcelEvent::new(0x0A, 0x0B, 999, 4);
491        assert!(ev.objects.is_empty());
492        assert_eq!(
493            (ev.event_code, ev.activity, ev.timestamp, ev.status),
494            (0x0A, 0x0B, 999, 4)
495        );
496
497        let ev = OcelEvent::from(0x1234u16);
498        assert_eq!(
499            (ev.event_code, ev.activity, ev.timestamp, ev.status),
500            (0x1234, 0, 0, 0)
501        );
502        assert!(ev.objects.is_empty());
503
504        let mut ev = OcelEvent::new(7, 9, 11, 0);
505        ev.objects.push(0xAA, 111);
506        ev.objects.push(0xBB, 222);
507        ev.objects.push(0xCC, 333);
508        assert_eq!(
509            ev.objects.as_slice(),
510            &[(0xAA, 111), (0xBB, 222), (0xCC, 333)]
511        );
512
513        // write_to: header-only, big-endian layout
514        let ev = OcelEvent::new(0x0102, 0x0304, 0x0506_0708_090a_0b0c, 4);
515        assert_eq!(ev.encoded_len(), OcelEvent::HEADER_BYTES);
516        let mut buf = [0u8; 64];
517        let n = ev.write_to(&mut buf).unwrap();
518        assert_eq!(n, OcelEvent::HEADER_BYTES);
519        assert_eq!(&buf[0..2], &0x0102u16.to_be_bytes());
520        assert_eq!(&buf[2..4], &0x0304u16.to_be_bytes());
521        assert_eq!(&buf[4..12], &0x0506_0708_090a_0b0cu64.to_be_bytes());
522        assert_eq!((buf[12], buf[13]), (4, 0)); // status, object count
523
524        // write_to: deterministic with objects
525        let mut ev = OcelEvent::new(7, 9, 11, 4);
526        ev.objects.push(0x00aa, 0x0102_0304_0506_0708);
527        ev.objects.push(0x00bb, 0x1112_1314_1516_1718);
528        let expected = OcelEvent::HEADER_BYTES + 2 * OcelEvent::OBJECT_BYTES;
529        assert_eq!(ev.encoded_len(), expected);
530        let mut a = [0u8; 64];
531        let mut b = [0u8; 64];
532        let na = ev.write_to(&mut a).unwrap();
533        let nb = ev.write_to(&mut b).unwrap();
534        assert_eq!(na, expected);
535        assert_eq!(a[..na], b[..nb]);
536        assert_eq!(a[13], 2);
537        assert_eq!(&a[14..16], &0x00aau16.to_be_bytes());
538        assert_eq!(&a[16..24], &0x0102_0304_0506_0708u64.to_be_bytes());
539
540        // write_to: rejects too-small buffer without writing
541        let mut ev = OcelEvent::new(1, 2, 3, 4);
542        ev.objects.push(5, 6);
543        let mut small = [0u8; 8];
544        assert_eq!(ev.write_to(&mut small), Err(EvidenceError::BufferTooSmall));
545        assert_eq!(small, [0u8; 8]);
546        let mut exact = [0u8; OcelEvent::HEADER_BYTES + OcelEvent::OBJECT_BYTES];
547        assert!(ev.write_to(&mut exact).is_ok());
548
549        // EvidenceError Copy + PartialEq
550        let e = EvidenceError::BufferTooSmall;
551        assert_eq!(e, EvidenceError::BufferTooSmall);
552        assert_eq!(e, e); // Copy
553    }

556    #[test]
557    fn ocel_log_and_json() {
558        // collection basics
559        let mut log = OcelLog::new();
560        assert!(log.is_empty());

567
568        // empty log renders []
569        assert_eq!(OcelLog::new().to_json(), "[]");
570
571        // JSON is object-centric with scalar fields and linked objects
572        let mut a = OcelEvent::new(10, 20, 30, 4);
573        a.objects.push(100, 1000);
574        a.objects.push(101, 1001);
575        let mut log = OcelLog::new();
576        log.push(a);
577        log.push(OcelEvent::new(11, 21, 31, 7));
578        let json = log.to_json();
579        assert!(json.starts_with('[') && json.ends_with(']'));
580        assert!(json.contains("\"event_code\":10"));
581        assert!(json.contains("\"activity\":20"));
582        assert!(json.contains("\"timestamp\":30"));
583        assert!(json.contains("\"status\":4"));
584        assert!(
585            json.contains("\"objects\":[{\"type\":100,\"id\":1000},{\"type\":101,\"id\":1001}]")
586        );
587        assert!(json.contains("\"status\":7,\"objects\":[]"));
588
589        // multiple objects per event are all present in JSON
590        let mut ev = OcelEvent::new(42, 1, 9999, 0);
591        ev.objects.push(0x01, 100);
592        ev.objects.push(0x02, 200);
593        ev.objects.push(0x03, 300);
594        let mut log = OcelLog::new();

596        let json = log.to_json();
597        assert!(json.contains("\"type\":1,\"id\":100"));
598        assert!(json.contains("\"type\":2,\"id\":200"));
599        assert!(json.contains("\"type\":3,\"id\":300"));
600    }
