19/// CommandOpcodeEncoded kernel. Packed-u64 ABI:
20/// - `state`: bits[0..8] = semantic command type (0..=127).
21/// - `input`: unused.
22///
23/// Returns: bits[0..8] = platform opcode class (0..3); bits[8..16] = original command type.
24#[inline]
25#[must_use]
26pub fn command_opcode_encoded(state: u64, input: u64) -> u64 {
27    let _ = input;
28    let cmd = (state & 0x7F) as u32;
29    // Branchless boundary classification: count how many thresholds cmd >= threshold.
30    // lt_mask_u32(cmd, boundary) = 0xFFFFFFFF when cmd < boundary, else 0.
31    // Negate and shift to get 1 when cmd >= boundary.
32    let ge32 = (!lt_mask_u32(cmd, 32) >> 31) as u32; // 1 if cmd >= 32
33    let ge64 = (!lt_mask_u32(cmd, 64) >> 31) as u32; // 1 if cmd >= 64
34    let ge96 = (!lt_mask_u32(cmd, 96) >> 31) as u32; // 1 if cmd >= 96
35                                                     // opcode_class = number of thresholds crossed = 0, 1, 2, or 3.
36    let opcode_class = ge32 + ge64 + ge96;
37    (opcode_class as u64) | ((cmd as u64) << 8)
38}
