16/// ConditionFlagEvaluated kernel. Packed-u64 ABI:
17/// - `state`: bits[0..32] = condition flags bitset (32 flags).
18/// - `input`: bits[0..8] = flag index to test (0..31).
19///
20/// Returns: bits[0..8] = 1 if flag is set, 0 otherwise; bits[8..16] = rank of flag among set flags.
21#[inline]
22#[must_use]
23pub fn condition_flag_evaluated(state: u64, input: u64) -> u64 {
24    let flags = state & 0xFFFF_FFFF;
25    let idx = input & 0x1F; // 5-bit index; safe to shift
26    let bit_set = (flags >> idx) & 1;
27    // Count set bits strictly below idx (rank_u64 is inclusive so use mask directly).
28    let rank = (flags & (1u64 << idx).wrapping_sub(1)).count_ones() as u64;
29    bit_set | (rank << 8)
30}
