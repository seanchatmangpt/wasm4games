18/// CtqThresholdEvaluated kernel. Packed-u64 ABI:
19/// - `state`: bits[0..16] = measured value (u16).
20/// - `input`: bits[0..16] = LSL (lower spec limit), bits[16..32] = USL (upper spec limit).
21///
22/// Returns: bits[0..8] = in_spec (1 if LSL <= value <= USL, else 0);
23///          bits[8..16] = direction (0=below LSL or in-spec, 1=above USL).
24#[inline]
25#[must_use]
26pub fn ctq_threshold_evaluated(state: u64, input: u64) -> u64 {
27    let value = (state & 0xFFFF) as u32;
28    let lsl = (input & 0xFFFF) as u32;
29    let usl = ((input >> 16) & 0xFFFF) as u32;
30
31    // above_lsl: all-ones when value >= lsl (i.e., NOT value < lsl)
32    let below_lsl = lt_mask_u32(value, lsl);
33    let above_lsl = !below_lsl;
34    // below_usl: all-ones when value <= usl (i.e., NOT usl < value)
35    let above_usl_mask = lt_mask_u32(usl, value);
36    let below_usl = !above_usl_mask;
37
38    // in_spec: 1 iff value >= lsl AND value <= usl
39    let in_spec = ((above_lsl & below_usl) >> 31) as u64;
40    // direction: 1 iff value > usl (above upper spec limit)
41    let above_usl = (above_usl_mask >> 31) as u64;
42
43    (in_spec & 0xFF) | ((above_usl & 0xFF) << 8)
44}
