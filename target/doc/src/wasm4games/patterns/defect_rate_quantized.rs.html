18/// DefectRateQuantized kernel. Packed-u64 ABI:
19/// - `state`: bits[0..32] = raw defect count (u32).
20/// - `input`: bits[0..32] = sample size (u32, treated as >= 1).
21///
22/// Returns: bits[0..32] = defect rate per million (0..=1_000_000).
23#[inline]
24#[must_use]
25pub fn defect_rate_quantized(state: u64, input: u64) -> u64 {
26    let defects = state & 0xFFFF_FFFF;
27    // Ensure sample >= 1 branchlessly: (sample | (sample == 0) as u64) gives 1 when 0.
28    let raw_sample = input & 0xFFFF_FFFF;
29    let is_zero = ((raw_sample == 0) as u64).wrapping_neg(); // all-ones if zero
30    let sample = raw_sample | (is_zero & 1u64); // substitute 1 when zero
31    let rate = (defects.saturating_mul(1_000_000)) / sample;
32    let clamped = clamp_u32(rate.min(1_000_000) as u32, 0, 1_000_000);
33    (clamped as u64) & 0xFFFF_FFFF
34}
