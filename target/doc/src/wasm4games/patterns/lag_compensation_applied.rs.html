18/// LagCompensationApplied kernel. Packed-u64 ABI:
19/// - `state`: bits[0..16] = server_position (u16), bits[16..32] = velocity (i16 as u16, two's complement).
20/// - `input`: bits[0..8] = lag_ticks (u8, number of ticks to rewind).
21///
22/// Returns: bits[0..16] = compensated position (clamped to [0, 0xFFFF]).
23#[inline]
24#[must_use]
25pub fn lag_compensation_applied(state: u64, input: u64) -> u64 {
26    let server_pos = (state & 0xFFFF) as i64;
27    let vel_raw = ((state >> 16) & 0xFFFF) as i64;
28    let velocity = (vel_raw as i16) as i64;
29    let lag = (input & 0xFF) as i64;
30    let rewind = saturating_sub_i64(server_pos, velocity * lag);
31    let clamped = clamp_u32(rewind.max(0) as u32, 0, 0xFFFF);
32    (clamped as u64) & 0xFFFF
33}
