48/// NavStateAdvanced kernel. Packed-u64 ABI:
49/// - `state`: bits[0..8] = current nav state (0=IDLE, 1=MOVING, 2=ARRIVED, 3=BLOCKED)
50/// - `input`: bits[0..8] = event symbol (0=STOP, 1=MOVE, 2=ARRIVE, 3=OBSTACLE)
51///
52/// Returns bits[0..8] = next nav state.
53#[inline]
54#[must_use]
55pub fn nav_state_advanced(state: u64, input: u64) -> u64 {
56    let current = (state & 0xFF) as usize;
57    let symbol = (input & 0xFF) as u8;
58    dfa_advance(current, symbol, &NAV_TABLE, 4) as u64
