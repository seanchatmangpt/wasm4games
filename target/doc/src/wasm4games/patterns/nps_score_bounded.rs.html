19/// NpsScoreBounded kernel. Packed-u64 ABI:
20/// - `state`: bits[0..8] = raw NPS response score (0..=10).
21/// - `input`: unused.
22///
23/// Returns: bits[0..8] = clamped score (0..=10);
24///          bits[8..16] = category (0=Detractor 0–6, 1=Passive 7–8, 2=Promoter 9–10).
25#[inline]
26#[must_use]
27pub fn nps_score_bounded(state: u64, input: u64) -> u64 {
28    let _ = input;
29    let score = (state & 0xFF) as u32;
30    let clamped = clamp_u32(score, 0, 10);
31
32    // Thresholds: Detractor <7, Passive 7..<9, Promoter >=9
33    let below_7 = lt_mask_u32(clamped, 7);
34    let below_9 = lt_mask_u32(clamped, 9);
35
36    // category: 0 if <7, 1 if 7<=<9, 2 if >=9
37    let cat_ge_7 = select_u32(below_9, 1, 2); // 1 if <9, else 2
38    let category = select_u32(below_7, 0, cat_ge_7); // 0 if <7, else cat_ge_7
39
40    (clamped as u64) | ((category as u64) << 8)
41}
