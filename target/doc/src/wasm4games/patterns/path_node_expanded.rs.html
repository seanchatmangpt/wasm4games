17/// PathNodeExpanded kernel. Packed-u64 ABI:
18/// - `state`: bits[0..32] = current node f-cost (u32)
19/// - `input`: bits[0..32] = candidate node f-cost, bits[32..48] = candidate node id
20///
21/// Returns bits[0..32] = winning f-cost, bits[32..48] = winning node id.
22#[inline]
23#[must_use]
24pub fn path_node_expanded(state: u64, input: u64) -> u64 {
25    let curr_cost = (state & 0xFFFF_FFFF) as u32;
26    let cand_cost = (input & 0xFFFF_FFFF) as u32;
27    let cand_id = ((input >> 32) & 0xFFFF) as u32;
28    // curr_id packed in state bits[32..48]; default to 0 if not provided
29    let curr_id = ((state >> 32) & 0xFFFF) as u32;
30    // mask = all-ones when cand_cost < curr_cost (candidate wins), else all-zeros
31    let cand_wins = lt_mask_u32(cand_cost, curr_cost);
32    let win_cost = select_u32(cand_wins, cand_cost, curr_cost);
33    let win_id = select_u32(cand_wins, cand_id, curr_id);
34    (win_cost as u64) | ((win_id as u64) << 32)
35}
