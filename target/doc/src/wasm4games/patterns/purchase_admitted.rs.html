45/// PurchaseAdmitted kernel. Packed-u64 ABI:
46/// - `state`: bits[0..8] = current purchase state (0..4, clamped to `[0, 5)`).
47/// - `input`: bits[0..8] = event symbol (0..4, clamped to `[0, 5)`).
48///
49/// Returns the next purchase state in bits[0..8].
50#[inline]
51#[must_use]
52pub fn purchase_admitted(state: u64, input: u64) -> u64 {
53    let st = ((state & 0xFF) as usize).min(FAILED);
54    let sym_raw = (input & 0xFF) as usize;
55    // Branchless clamp: OOB symbols map to ADD (0), holding the default transition.
56    let in_alphabet = (sym_raw < ALPHABET) as usize;
57    let sym = (sym_raw & 0usize.wrapping_sub(in_alphabet)) as u8;
58    let next = dfa_advance(st, sym, &TABLE, ALPHABET);
59    (next as u64) & 0xFF
60}
