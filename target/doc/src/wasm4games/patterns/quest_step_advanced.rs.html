6//! # Branchless Contract
7//! **Ensures:** result matches the reference oracle for all inputs.
8//! **Invariant:** execution path is independent of input data values (Branchless).
9//!
10//! # Timing contract

13//! Advance a linear quest one step with [`bcinr_logic::dfa::dfa_advance`]: `complete` moves
14//! forward through steps 0..4 (Done is absorbing), `abandon` resets to step 0, `idle` holds.

16use bcinr_logic::dfa::dfa_advance;

18/// States: 0..4 = step index, 4 = Done (absorbing). Alphabet: 0=idle, 1=complete, 2=abandon.
19const ALPHABET: usize = 3;
20#[rustfmt::skip]
21const TABLE: [usize; 15] = [
22    // input:   idle complete abandon
23    /* step0 */ 0, 1, 0,
24    /* step1 */ 1, 2, 0,
25    /* step2 */ 2, 3, 0,
26    /* step3 */ 3, 4, 0,
27    /* Done  */ 4, 4, 4,
28];
29
30/// QuestStepAdvanced kernel. Packed-u64 ABI:
31/// - `state`: bits[0..8] = current step (clamped into `[0, 5)`).
32/// - `input`: bits[0..8] = quest symbol (clamped into `[0, 3)`).
33///
34/// Returns the next step in bits[0..8].
35#[inline]
36#[must_use]
37pub fn quest_step_advanced(state: u64, input: u64) -> u64 {
38    let st = ((state & 0xFF) as usize).min(4);
39    let sym_raw = (input & 0xFF) as usize;
40    // Branchless clamp of the symbol into [0, ALPHABET): dfa_advance returns 0 for OOB, but
41    // we want OOB symbols to act as `idle` (hold), so map them onto 0.
42    let in_alphabet = (sym_raw < ALPHABET) as usize;
43    let sym = (sym_raw & 0usize.wrapping_sub(in_alphabet)) as u8;
44    let next = dfa_advance(st, sym, &TABLE, ALPHABET);
45    (next as u64) & 0xFF
46}
