19/// ReplayFrameRecorded kernel. Packed-u64 ABI:
20/// - `state`: prior rolling replay digest.
21/// - `input`: bits[0..16] = tick, bits[16..40] = input word, bits[40..64] = state digest lo.
22///
23/// Returns the new rolling replay digest.
24#[inline]
25#[must_use]
26pub fn replay_frame_recorded(state: u64, input: u64) -> u64 {
27    let tick = input & 0xFFFF;
28    let in_word = (input >> 16) & 0xFF_FFFF;
29    let st_digest = (input >> 40) & 0xFF_FFFF;
30    let mut r = DeterministicSubstrateReceipt {
31        current_hash: state,
32        steps: tick,
33    };
34    r.record(in_word, st_digest, state);
35    r.finalize()
36}
