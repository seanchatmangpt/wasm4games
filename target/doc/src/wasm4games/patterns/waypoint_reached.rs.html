18/// WaypointReached kernel. Packed-u64 ABI:
19/// - `state`: bits[0..32] = current distance (u32, fixed-point units)
20/// - `input`: bits[0..32] = tolerance threshold
21///
22/// Returns bits[0..8] = 1 if reached (dist <= tol), 0 otherwise; bits[8..40] = saturating remaining = max(dist - tol, 0).
23#[inline]
24#[must_use]
25pub fn waypoint_reached(state: u64, input: u64) -> u64 {
26    let dist = (state & 0xFFFF_FFFF) as u32;
27    let tol = (input & 0xFFFF_FFFF) as u32;
28    // reached = 1 when dist <= tol, i.e. NOT (tol < dist)
29    let tol_lt_dist = lt_mask_u32(tol, dist);
30    let reached = ((!tol_lt_dist) & 1) as u64;
31    // remaining = max(dist - tol, 0) via saturating sub on i64
32    let remaining = saturating_sub_i64(dist as i64, tol as i64).max(0_i64) as u64 & 0xFFFF_FFFF;
33    reached | (remaining << 8)
34}
