1//! In-crate self-checks: an *offline* admissibility proxy.
2//!
3//! # What This Module Can and Cannot Verify
4//!
5//! **Can verify (offline):**
6//! - Status codes are in-bounds for the known [`crate::class::status`] vocabulary.
7//! - Replay frames produce a deterministic digest when folded twice with the same input.
8//! - Negative fixtures carry the correct refusal status for their pattern.
9//! - All items in [`crate::patterns::PATTERN_REGISTRY`] are internally self-consistent.
10//!
11//! **Cannot verify (requires `wasm4pm`):**
12//! - Whether a real authority would admit or refuse a given event.
13//! - Receipt chain integrity across process boundaries.
14//! - Scope-level conformance rules from the `wasm4pm-compat` ontology.
15//!
16//! **This module is NOT the admission authority.** The external `wasm4pm` service
17//! performs real admission/refusal (see [`crate::compat`]). Offline checks here give
18//! fast, dependency-free confidence that the registry is self-consistent and that
19//! individual pattern kernels behave deterministically, but they do not substitute for
20//! a live `wasm4pm` conformance check.
21

28/// A structured report from a single offline validation pass.
29///
30/// Collects the results of all checks performed by [`validate_pattern`] so that callers
31/// can inspect each finding individually rather than receiving a bare `bool`. Fields are
32/// `true` when the check passed.
33#[derive(Clone, Copy, PartialEq, Eq, Debug)]
34pub struct ValidationReport {
35    /// The pattern id that was checked.
36    pub pattern_id: u16,
37    /// `true` if the required status code is within the known vocabulary.
38    pub required_status_in_bounds: bool,
39    /// `true` if the refusal status code is within the known vocabulary.
40    pub refusal_status_in_bounds: bool,
41    /// `true` if the negative fixture carries the correct refusal status.
42    pub negative_fixture_status_correct: bool,
43    /// `true` if the negative fixture links to the correct activity id.
44    pub negative_fixture_activity_correct: bool,
45}
46
47impl ValidationReport {
48    /// `true` when all checks in this report passed.
49    #[inline]
50    #[must_use]
51    pub fn is_ok(&self) -> bool {
52        self.required_status_in_bounds
53            && self.refusal_status_in_bounds
54            && self.negative_fixture_status_correct
55            && self.negative_fixture_activity_correct
56    }
57}
58
59/// Run every offline check for a single pattern spec and return a [`ValidationReport`].
60///
61/// # Must Use
62///
63/// Ignoring the report silently discards potential conformance failures.
64#[must_use = "offline validation result — ignoring may miss conformance failures"]
65pub fn validate_pattern(spec: &PatternSpec) -> ValidationReport {
66    let nf = negative_fixture(spec);
67    ValidationReport {
68        pattern_id: spec.id.0,
69        required_status_in_bounds: check_status_bounds(spec, spec.admission.required_status),
70        refusal_status_in_bounds: check_status_bounds(spec, spec.admission.refusal_status),
71        negative_fixture_status_correct: nf.status == spec.admission.refusal_status,
72        negative_fixture_activity_correct: nf.activity == spec.id.0,
73    }
74}

76/// Check that an observed status code is within the known vocabulary.
77///
78/// Returns `true` if `observed < `[`crate::class::status::COUNT`].
79#[inline]
80#[must_use = "offline validation result — ignoring may miss conformance failures"]
81pub fn check_status_bounds(_spec: &PatternSpec, observed: u8) -> bool {
82    observed < status::COUNT
83}

85/// Fold replay frames into a rolling digest.
86///
87/// The digest is deterministic: identical frames always produce the same value.
88/// Use [`check_replay_determinism`] to verify this property at runtime.
89#[must_use = "offline validation result — ignoring may miss conformance failures"]
90pub fn replay_digest(frames: &[ReplayFrame]) -> u64 {
91    let mut r = DeterministicSubstrateReceipt::new();
92    for f in frames {
93        r.record(f.tick, f.input, f.state_digest);
94    }
95    r.finalize()
96}
97
98/// Check that replaying the same frames reproduces the same digest (determinism).
99///
100/// A `false` return would indicate a non-deterministic implementation, which violates
101/// the core replay-evidence contract.
102#[must_use = "offline validation result — ignoring may miss conformance failures"]
103pub fn check_replay_determinism(frames: &[ReplayFrame]) -> bool {
104    replay_digest(frames) == replay_digest(frames)
105}

107/// Construct a negative fixture for a pattern: an event whose status is the pattern's
108/// refusal code, which a correct authority MUST refuse.
109///
110/// Use this in integration tests against a live `wasm4pm` instance to verify that it
111/// correctly refuses events carrying the refusal status.
112#[must_use]
113pub fn negative_fixture(spec: &PatternSpec) -> OcelEvent {
114    OcelEvent::new(spec.event.code, spec.id.0, 0, spec.admission.refusal_status)
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120    use crate::evidence::replay::ReplayFrame;
121    use crate::ir::{AdmissionRule, EventKind, LoweringKind, ObjectKind, PatternId, PatternSpec};
122    use crate::patterns::PATTERN_REGISTRY;
123
124    #[test]
125    fn registry_validation_and_replay_determinism() {
126        // All registry specs must pass offline validation (status bounds + negative fixtures).
127        for spec in PATTERN_REGISTRY {
128            assert!(check_status_bounds(spec, spec.admission.refusal_status));
129            let ev = negative_fixture(spec);
130            assert_eq!(ev.status, spec.admission.refusal_status);
131            assert_eq!(ev.activity, spec.id.0);
132            let report = validate_pattern(spec);
133            assert!(
134                report.is_ok(),
135                "pattern id={} failed: {:?}",
136                spec.id.0,
137                report
138            );
139        }
140
141        // Replay determinism: non-empty and empty frames.
142        let frames = [
143            ReplayFrame::new(0, 0x0000_0001, 0xDEAD_BEEF),
144            ReplayFrame::new(1, 0x0000_0002, 0xCAFE_BABE),
145            ReplayFrame::new(2, 0x0000_0003, 0x1234_5678),
146        ];
147        assert!(check_replay_determinism(&frames));
148        assert!(check_replay_determinism(&[]));
149    }
150
151    #[test]
152    fn out_of_bounds_refusal_status_fails_validation() {
153        let bad_spec = PatternSpec {
154            id: PatternId(0xFFFF),
155            name: "bad_test_pattern",
156            lowering: LoweringKind::Mask,
157            state_card: 2,
158            event: EventKind {
159                code: 0x9999,
160                name: "BadEvent",
161            },
162            objects: &[ObjectKind {
163                code: 0x0001,
164                name: "player",
165            }],
166            admission: AdmissionRule {
167                required_status: 4,
168                refusal_status: 255,
169            },
170            otel_span: 0x9999,
171        };
172        let report = validate_pattern(&bad_spec);
173        assert!(!report.refusal_status_in_bounds);
174        assert!(!report.is_ok());
175    }
